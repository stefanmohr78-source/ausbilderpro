⚠️ Firestore-Regeln prüfen
Damit Ausbilder, Lehrgangsarten und Berufsgruppen gespeichert werden können, muss shared/kursSettings in den Firestore Security Rules erlaubt sein.
Firebase Console → Firestore → Regeln — ersetze die bestehenden Regeln durch: 
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAdmin() {
      return request.auth != null
        && (get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin'
         || get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'verwaltung');
    }

    match /users/{uid} {
      allow read, write: if request.auth != null && request.auth.uid == uid;
      allow read: if isAdmin();
      match /{sub}/{subDoc} {
        allow read, write: if request.auth != null && request.auth.uid == uid;
        allow read: if isAdmin();
      }
    }

    match /shared/{docId} {
      allow read: if request.auth != null;
      allow write: if isAdmin();
      match /{sub}/{subDoc} {
        allow read: if request.auth != null;
        allow write: if isAdmin();
        match /{deep}/{deepDoc} {
          allow read: if request.auth != null;
          allow write: if isAdmin();
          // Ausbilder darf TN-Status + Attendance-Aggregat aktualisieren
          allow update: if request.auth != null
            && request.resource.data.diff(resource.data).affectedKeys().hasOnly(
              ['status','attTotal','attPresent','attAbsentE','attAbsentU','attSick','attBerufsschule','attLate','attEarly','attPercent','attLastSync']
            );
        }
      }
    }

    match /globalTN/{doc} {
      allow read, write: if isAdmin();
      // Ausbilder darf nur status + wiederholer Felder ändern (Sync aus AusbilderPro)
      allow update: if request.auth != null
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['status', 'wiederholer']);
    }
  }
}
